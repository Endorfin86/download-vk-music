git config --global user.name 'NAME' # Глобальная настройка имени пользователя для аккаунта
git config --global user.email 'EMAIL' # Глобальная настройка электронной почты для аккаунта

git init # Инициализация проекта. Для Инициализации необходимо зайти в папку с проектом и ввести команду в терминале
git add . # Добавление файлов в загрузку на сервер
git rm --cached * # Очистить файлы на добавление на сервер
git commit -m 'COMMENT' # Закомитить файлы на сервер
git branch NAME # Создать ветку
git checkout NAME # Перейти на другую ветку
git marge NAME # Объеденить ветки. Для объединения перейти на ветку мастер в терминале вписать команду для объединения и вместо Name указать имя ветки для присоединения
